\frame{
\begin{block}{The Lotka- Volterra equations : }
\begin{equation*}
\begin{array}{r l}
x′ & = f(t,x,y)=x-xy- \frac{1}{10} x^2 \\
& \\
y′ & =g(t,x,y)=xy-y- \frac{1}{20} y^2,
\end{array}
\end{equation*}
with the initial condition $x(0) = 2$ and $y(0) = 1$ for $0 \le t \le 30$.
\end{block}
